# 倒计时

let count = 3
let loading = Loading.create({ text: `${count}s` })
let timer = setInterval(() => {
	count--
	if (count === 0) {
		clearInterval(timer)
		loading.remove()
	} else {
		loading.update({ text: `${count}s` })
	}
}, 1000);
