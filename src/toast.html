# Toast 提示

页面toast提示。

# 基本用法

# 组件引入

import { Toast } from 'rong-ui2';
1

# 函数调用

直接使用Toast.create({options})

Toast.create('长文本形式为文字超过十个时使用，宽度固定')
1

# API

参数 说明 类型 可选值 默认值
content 内容 ReactNode
type icon类型 String long/positive/negetive long
time 时长 number 2000
callback 回调 function

# Method

  • Toast.positive('')
  • Toast.negetive('')

# 全局配置Toast

import { Toast } from 'rong-ui2';

Toast.config({ time: 3000 })
