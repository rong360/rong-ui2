# FlexFixed 布局

header和footer固定，content内容滚动

# 基本用法

# 组件引入

import { Flexfixed } from 'rong-ui2';
1

# 调用

return <FlexFixed>content</FlexFixed>
1

# API

参数 说明 类型 可选值 默认值
className 自定义 CSS class string
header 页头 ReactNode
footer 页尾 ReactNode
extra 额外插槽
位置在footer之后		 ReactNode
onScroll 页面滚动回调 function({scrollTop, maxScrollTop, direction})
onScrollDebounce 页面滚动停止之后回调 function({scrollTop, maxScrollTop, direction})
hideFooterOnKeyboardShow 键盘弹起之后隐藏footer boolean true
# 特别提示：嵌套overflow-x:scroll 或 overflow-y:scroll滚动元素时需在滚动元素上添加scroll-area样式，如：
<div style="overflow-x:scroll" class="scroll-area"></div>
1

代码演示



const goHome = useCallback(() => {}, [])
const onScroll = useCallback(({scrollTop, maxScrollTop, direction}) => {}, [])
const onScrollDebounce = useCallback(({scrollTop, maxScrollTop, direction}) => {
  if( (maxScrollTop-scrollTop) < 10 ){
      console.log('滚动底啦~')
  }
}, [])

const header = <Titlebar theme='b' onBack={goHome}>FlexFixed</Titlebar>
const footer = <Button>提交</Button>
const extra = <div style={{position: 'absolute', bottom: 0, right: 0, background: 'rgba(0,0,0, 0.7)', color: '#fff', margin: 0}}>
  this is extra
</div>

return <FlexFixed 
  header={header} 
  footer={footer} 
  extra={extra} 
  onScroll={onScroll} 
  onScrollDebounce={onScrollDebounce}
  >
  this is content
</FlexFixed>
